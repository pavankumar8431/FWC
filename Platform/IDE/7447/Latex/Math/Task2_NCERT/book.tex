% ===================== EXERCISE 6.1 =====================
\exercise{6.1}

\begin{enumerate}
\item Find the rate of change of the area of a circle with respect to its radius $r$ when
\begin{enumerate}
\item $r=3$ cm
\item $r=4$ cm
\end{enumerate}

\item The volume of a cube is increasing at the rate of $8\text{ cm}^3/\text{s}$.  
How fast is the surface area increasing when the edge is $12$ cm?

\item The radius of a circle is increasing uniformly at $3$ cm/s.  
Find the rate at which the area is increasing when the radius is $10$ cm.

\item An edge of a cube is increasing at $3$ cm/s.  
How fast is the volume increasing when the edge is $10$ cm?

\item Waves move in circles at $5$ cm/s.  
When the radius is $8$ cm, how fast is the enclosed area increasing?
\end{enumerate}

\end{enumerate}
